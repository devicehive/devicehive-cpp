643  /// @brief Update a device from the JSON value.
644  /**
645  @param[in] jval The JSON value.
646  @param[in,out] device The device to update.
647  */
648  static void fromJson(json::Value const& jval, DevicePtr device)
649  {
650  try
651  {
652  { // just check identifier is the same!
653  const String id = jval["id"].asString();
654  if (id.empty())
655  throw std::runtime_error("identifier is empty");
656  if (!boost::iequals(device->id, id))
657  throw std::runtime_error("invalid identifier");
658  }
659 
660  device->name = jval["name"].asString();
661  //device->key = jval["key"].asString();
662  device->status = jval["status"].asString();
663  device->data = jval["data"];
664 
665  { // update network
666  json::Value const& jnet = jval["network"];
667  if (!jnet.isNull())
668  {
669  if (!device->network)
670  device->network = Network::create(String(), String());
671  fromJson(jnet, device->network);
672  }
673  }
674 
675  { // update device class
676  json::Value const& jclass = jval["deviceClass"];
677  if (!jclass.isNull())
678  {
679  if (!device->deviceClass)
680  device->deviceClass = Device::Class::create(String(), String());
681  fromJson(jclass, device->deviceClass);
682  }
683  }
684 
685  { // equipment
686  json::Value const& json_eq = jval["equipment"];
687  if (!json_eq.isNull() && !json_eq.isArray())
688  throw std::runtime_error("equipment is not an array");
689 
690  const size_t N = json_eq.size();
691  for (size_t i = 0; i < N; ++i)
692  equipmentFromJson(json_eq[i], device);
693  }
694  }
695  catch (std::exception const& ex)
696  {
697  OStringStream ess;
698  ess << "failed to update Device:\n"
699  << ex.what();
700  throw std::runtime_error(ess.str().c_str());
701  }
702  }

712  const size_t N = device->equipment.size();
713  json::Value json_eq; json_eq.resize(N);
714  for (size_t i = 0; i < N; ++i)
715  json_eq[i] = toJson(device->equipment[i]);

719  if (!device->name.empty())

721  jval["key"] = device->key;

723  if (device->network)
724  jval["network"] = toJson(device->network);
725  if (device->deviceClass)
