577  /// @brief Send the gateway frame.
578  /**
579  @param[in] frame The gateway frame to send.
580  @param[in] da64 The destination MAC address.
581  @param[in] da16 The destination network address.
582  */
583  void sendGatewayFrame(gateway::Frame::SharedPtr frame, UInt64 da64, UInt16 da16)
584  {
585  if (frame && !frame->empty())
586  {
587  const size_t len = frame->size();
588  if (XBEE_FRAGMENTATION < len) // do fragmentation
589  {
590  std::vector<UInt8>::const_iterator data = frame->getContent().begin();
591  const size_t N = len/XBEE_FRAGMENTATION - 1; // number of "full" frames
592 
593  for (size_t i = 0 ; i < N; ++i)
594  {
595  xbee::Frame::ZBTransmitRequest payload(String(data, data + XBEE_FRAGMENTATION), m_xbeeFrameId++, da64, da16);
596  sendXBeeFrame(payload);
597  data += XBEE_FRAGMENTATION;
598  }
599 
600  // send the rest of frames
601  const size_t rest = (len - N*XBEE_FRAGMENTATION);
602  const size_t len1 = rest/2;
603  const size_t len2 = rest - len1;
604 
605  xbee::Frame::ZBTransmitRequest payload1(String(data, data + len1), m_xbeeFrameId++, da64, da16);
606  data += len1;
607 
608  xbee::Frame::ZBTransmitRequest payload2(String(data, data + len2), m_xbeeFrameId++, da64, da16);
609  data += len2;
610 
611  sendXBeeFrame(payload1);
612  sendXBeeFrame(payload2);
613  }
614  else // just one frame
615  {
616  const std::vector<UInt8> &data = frame->getContent();
617  xbee::Frame::ZBTransmitRequest payload(String(data.begin(), data.end()), m_xbeeFrameId++, da64, da16);
618  sendXBeeFrame(payload);
619  }
620  }
621  }

716  /// @brief Handle the received frame.
717  /**
718  @param[in] frame The frame.
719  */
720  void handleRecvXBeeFrame(xbee::Frame::SharedPtr frame)
721  {
722  switch (frame->getIntent())
723  {
724  case xbee::Frame::ZB_RECEIVE_PACKET:
725  {
726  xbee::Frame::ZBReceivePacket payload;
727  if (frame->getPayload(payload))
728  {
729  boost::shared_ptr<ZDevice> zdev = getZDevice(payload.srcAddr64);
730  zdev->address16 = payload.srcAddr16;
731 
732  { // put the data to the stream buffer
733  OStream os(&zdev->m_rx_buf);
734  os.write(payload.data.data(), payload.data.size());
735  }
736 
737  HIVELOG_DEBUG(m_log, "got [" << dump::hex(payload.data) << "] from "
738  << dump::hex(payload.srcAddr64) << "/" << dump::hex(payload.srcAddr16));
739 
740  gateway::Frame::ParseResult result = gateway::Frame::RESULT_SUCCESS;
741  if (gateway::Frame::SharedPtr frame = gateway::Frame::parseFrame(zdev->m_rx_buf, &result))
742  {
743  handleGatewayMessage(frame->getIntent(),
744  zdev->gw.frameToJson(frame), zdev);
745  }
746  }
747  } break;
748 
749  default:
750  // do nothing
751  break;
752  }
753  }

757  /**
758  @param[in] zdev The ZigBee device.
759  @param[in] jdev The JSON device description.
760  */

762  {
763  { // create device
764  String id = jdev["id"].asString();
765  String key = jdev["key"].asString();
766  String name = jdev["name"].asString();
767

769  devicehive::Serializer::fromJson(jdev["deviceClass"], deviceClass);
770 
771  zdev->device = devicehive::Device::create(id, name, key, deviceClass, m_network);
772  zdev->device->status = "Online";
773 
774  zdev->delayedNotifications.clear();
775  zdev->deviceRegistered = false;
776  }
777 
778  { // update equipment
779  json::Value const& equipment = jdev["equipment"];
780  const size_t N = equipment.size();
781  for (size_t i = 0; i < N; ++i)
782  {

790  }
791  }

795  /// @brief Handle the incomming message.
796  /**
797  @param[in] intent The message intent.
798  @param[in] data The custom message data.
799  @param[in] zdev The ZigBee device.
800  */
801  void handleGatewayMessage(int intent, json::Value const& data, ZDeviceSPtr zdev)
802  {
803  HIVELOG_INFO(m_log, "process intent #" << intent << " data: " << data << "\n");
804 
805  if (intent == gateway::INTENT_REGISTRATION_RESPONSE)
806  {
807  zdev->gw.handleRegisterResponse(data);
808  createDevice(zdev, data);
809  m_service->asyncRegisterDevice(zdev->device);
810  }
811 
812  else if (intent == gateway::INTENT_REGISTRATION2_RESPONSE)
813  {
814  json::Value jdev = json::fromStr(data["json"].asString());
815  zdev->gw.handleRegister2Response(jdev);
816  createDevice(zdev, jdev);
817  m_service->asyncRegisterDevice(zdev->device);
818  }
819 
820  else if (intent == gateway::INTENT_COMMAND_RESULT_RESPONSE)
821  {
822  if (zdev->device)
823  {
824  HIVELOG_DEBUG(m_log, "got command result");
825 
826  devicehive::CommandPtr command = devicehive::Command::create();
827  command->id = data["id"].asUInt();
828  command->status = data["status"].asString();
829  command->result = data["result"].asString();
830 
831  m_service->asyncUpdateCommand(zdev->device, command);
832  }
833  else
834  HIVELOG_WARN_STR(m_log, "got command result before registration, ignored");
835  }
836 
837  else if (intent >= gateway::INTENT_USER)
838  {
839  if (zdev->device)
840  {
841  String name = zdev->gw.findNotificationNameByIntent(intent);
842  if (!name.empty())
843  {
844  HIVELOG_DEBUG(m_log, "got notification");
845 
846  devicehive::NotificationPtr notification = devicehive::Notification::create(name, data["parameters"]);
847 
848  if (zdev->deviceRegistered)
849  {
850  m_service->asyncInsertNotification(zdev->device, notification);
851  }
852  else
853  {
854  HIVELOG_DEBUG_STR(m_log, "device is not registered, notification delayed");
855  zdev->delayedNotifications.push_back(notification);
856  }
857  }
858  else
859  HIVELOG_WARN(m_log, "unknown notification: " << intent << ", ignored");
860  }
861  else
862  HIVELOG_WARN_STR(m_log, "got notification before registration, ignored");
863  }
864 
865  else
866  HIVELOG_WARN(m_log, "invalid intent: " << intent << ", ignored");
867  }
