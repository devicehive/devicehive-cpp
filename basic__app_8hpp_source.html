33 /// @brief The custom delayed task.
34 class DelayedTask
35 {
36  typedef DelayedTask This; ///< @brief The type alias.
37 
38 protected:
39 
40  /// @brief The main constructor.
41  /**
42  @param[in] ios The IO service.
43  */
44  DelayedTask(boost::asio::io_service &ios)
45  : m_timer(ios)
46  , m_started(false)
47  , m_cancelled(false)
48  {}
49 
50 public:
51 
52  /// @brief The trivial destructor.
53  virtual ~DelayedTask()
54  {}
55 
56 public:
57 
58  /// @brief The shared pointer type.
59  typedef boost::shared_ptr<DelayedTask> SharedPtr;
60 
61  /// @brief The factory method.
62  /**
63  @param[in] ios The IO service.
64  @param[in] timeout_ms The timeout, milliseconds.
65  @return The delayed task instance.
66  */
67  static SharedPtr create(boost::asio::io_service &ios, long timeout_ms)
68  {
69  SharedPtr pthis(new This(ios));
70 
71  if (0 < timeout_ms)
72  {
73  pthis->m_timer.expires_from_now(boost::posix_time::milliseconds(timeout_ms));
74  pthis->m_timer.async_wait(
75  boost::bind(&This::onTimer, pthis,
76  boost::asio::placeholders::error));
77  pthis->m_started = true;
78  }
79  else
80  {
81  // call as soon as possible
82  ios.post(boost::bind(&This::onTimer, pthis,
83  boost::system::error_code()));
84  }
85 
86  return pthis;
87  }
88 
89 
90  /// @brief Cancel the task.
91  /**
92  This method cancels current task.
93  */
94  void cancel()
95  {
96  m_cancelled = true;
97  if (m_started)
98  m_timer.cancel();
99  }
100 
101 public:
102 
103  /// @brief The callback type.
104  typedef boost::function0<void> Callback;
105 
106 
107  /// @brief Call this method when task is done.
108  /**
109  This callback will be called when this task is finished (successful or not).
110 
111  @param[in] cb The callback to call.
112  */
113  void callWhenDone(Callback cb)
114  {
115  if (!m_callback)
116  m_callback = cb;
117  else
118  m_callback = boost::bind(&This::tie, m_callback, cb);
119  }
120 
121 private:
122 
123  /// @brief Call two callbacks.
124  /**
125  @param[in] cb1 The first callback.
126  @param[in] cb2 The second callback.
127  */
128  static void tie(Callback cb1, Callback cb2)
129  {
130  cb1();
131  cb2();
132  }
133 
134 private:
135 
136  /// @brief Run the delayed task.
137  void onTimer(boost::system::error_code err)
138  {
139  if (!err && !m_cancelled)
140  {
141  if (Callback cb = m_callback)
142  {
143  m_callback = Callback();
144  cb(); // call the method!
145  }
146  }
147  else
148  m_callback = Callback();
149  }
150 
151 private:
152  boost::asio::deadline_timer m_timer; ///< @brief The deadline timer.
153  boost::function0<void> m_callback; ///< @brief The callback method.
154 
155  bool m_started; ///< @brief The timer "started" flag.
156  bool m_cancelled; ///< @brief The "cancelled" flag.
157 };

420  /// @brief The signal handler.
421  /**
422  @param[in] err The error code.
423  @param[in] signo The signal number.
424  */
425  void onGotSignal(boost::system::error_code err, int signo)
426  {
427  if (!err)
428  {
429  HIVELOG_INFO(m_log, "got "
430  << getSignalName(signo)

432 
433  switch (signo)
434  {

436  case SIGINT:
437  m_terminated += 1; // force 'terminated' flag
438  stop(); // since stop() is virtual
439  break;

442  break;
443  }

451  else
