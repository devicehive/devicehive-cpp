481  ///@brief The frame parse result.
482  enum ParseResult
483  {

485  RESULT_INCOMPLETE, ///< @brief Need more data.

487  RESULT_BADSIGNATURE, ///< @brief Bad signature.
488  RESULT_BADVERSION ///< @brief Bad protocol version.
489  };

491 
492  /// @brief Assign the frame content.
493  /**
494  This method searches frame signature, veryfies the checksum
495  and assigns the frame content.
496 
497  @param[in] first The begin of input data.
498  @param[in] last The end of input data.
499  @param[out] n_skip The number of bytes processed.
500  @param[in,out] result The parse result. May be NULL.
501  @return Parsed frame or NULL.
502  */
503  template<typename In>
504  static SharedPtr parseFrame(In first, In last, size_t &n_skip, ParseResult *result)
505  {
506  ParseResult res = RESULT_INCOMPLETE;
507  SharedPtr frame; // empty frame
508 
509  // search for signature
510  for (; first != last; ++first, ++n_skip)
511  {
512  if (UInt8(*first) == SIGNATURE1)
513  break;
514  }
515 
516  const size_t buf_len = std::distance(first, last);
517  if (HEADER_LEN+FOOTER_LEN <= buf_len)
518  {
519  const In s_first = first;
520 
521  ++first; // skip signature1
522  if (UInt8(*first++) == SIGNATURE2) // check signature2
523  {
524  if (UInt8(*first++) == VERSION) // check version
525  {
526  ++first; // skip flags
527  const int len_lsb = UInt8(*first++);
528  const int len_msb = UInt8(*first++);
529  const size_t len = (len_msb<<8) | len_lsb;
530  ++first; ++first; // skip intent
531 
532  if (size_t(HEADER_LEN+len+FOOTER_LEN) <= buf_len)
533  {
534  if (checksum(s_first, s_first+(HEADER_LEN+len+FOOTER_LEN)) != 0x00)
535  {
536  // bad checksum
537  res = RESULT_BADCHECKSUM;
538  n_skip += 1; // skip signature (first byte)
539  }
540  else
541  {
542  // assign the frame content
543  frame.reset(new Frame());
544  frame->m_content.assign(s_first,
545  s_first+(HEADER_LEN+len+FOOTER_LEN));
546  n_skip += HEADER_LEN+len+FOOTER_LEN;
547  res = RESULT_SUCCESS;
548  }
549  }
550  // else incomplete
551  }
552  else
553  {
554  // unsupported version
555  res = RESULT_BADVERSION;
556  n_skip += 1; // skip signature (first byte)
557  }
558  }
559  else
560  {
561  // bad signature
562  res = RESULT_BADSIGNATURE;
563  n_skip += 1; // skip signature (first byte)
564  }
565  }
566  // else incomplete
567 
568  if (result)
569  *result = res;
570  return frame;
571  }
572 
573 
574  /// @brief Assign the frame content from stream buffer.
575  /**
576  This method searches frame signature, veryfies the checksum
577  and assigns the frame content.
578 
579  @param[in,out] sb The input data buffer.
580  @param[in,out] result The parse result. May be NULL.
581  @return Parsed frame or NULL.
582  */
583  static SharedPtr parseFrame(boost::asio::streambuf &sb, ParseResult *result)
584  {
585  const boost::asio::streambuf::const_buffers_type bufs = sb.data();
586  size_t n_skip = 0; // number of bytes to skip
587  SharedPtr frame = parseFrame(boost::asio::buffers_begin(bufs),
588  boost::asio::buffers_end(bufs), n_skip, result);
589  sb.consume(n_skip);
590  return frame;
591  }

604

617  /**

619  @param[in] last The end of frame data payload.
620  @return The checksum byte.
621  */
622  template<typename In>
623  static UInt8 checksum(In first, In last)
624  {
625  int cs = 0;
626  for (; first != last; ++first)
627  cs += UInt8(*first);
628  return UInt8(0xFF - (cs&0xFF));
629  }

632  /// @brief Initialize frame content from payload data.
633  /**
634  @param[in] intent The payload intent.
635  @param[in] payload The frame payload data.
636  */
637  void init(int intent, String const& payload)
638  {
639  const size_t length = payload.size();
640  assert(length < 64*1024 && "frame data payload too big");
641 
642  m_content.reserve(HEADER_LEN+length+FOOTER_LEN);
643 
644  // header (signature+intent+length)
645  m_content.push_back(SIGNATURE1);
646  m_content.push_back(SIGNATURE2);
647  m_content.push_back(VERSION);
648  m_content.push_back(0); // flags
649  m_content.push_back((length)&0xFF); // length, LSB
650  m_content.push_back((length>>8)&0xFF); // length, MSB
651  m_content.push_back((intent)&0xFF); // intent, LSB
652  m_content.push_back((intent>>8)&0xFF); // intent, MSB
653 
654  // payload
655  m_content.insert(m_content.end(),
656  payload.begin(), payload.end());
657 
658  // checksum
659  m_content.push_back(checksum(
660  m_content.begin(),
661  m_content.end()));
662  }
