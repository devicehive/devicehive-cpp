406  virtual void onPongTimedOut(ErrorCode err)
407  {
408  if (!err)
409  {
410  HIVELOG_WARN(m_log, "PONG timeout attempt #" << m_pingPong.ping_attempt
411  << " of " << m_pingPong.ping_retry_limit);
412 
413  if (m_pingPong.ping_attempt >= m_pingPong.ping_retry_limit)
414  {
415  HIVELOG_ERROR_STR(m_log, "no more attempts, close the connection");
416 
417  if (m_actionReceivedCallback)
418  m_ios.post(boost::bind(m_actionReceivedCallback, boost::asio::error::timed_out, json::Value()));
419  else
420  HIVELOG_WARN_STR(m_log, "no action callback, ignored");
421 
422  close(true);
423  }
424  else
425  {
426  asyncSendPingFrame();
427  startPingPongTimeout();
428  }
429  }
430  else if (err == boost::asio::error::operation_aborted)
431  {
432  // do nothing
433  }
434  else
435  {
436  HIVELOG_ERROR(m_log, "ping/pong timeout error ["
437  << err << "] " << err.message());
438  }
439  }

761  assert(!"callback is dead or not initialized");
762  }

766  /// @brief The "connected" handler.
767  /**
768  @param[in] err The error code.
769  */
770  void onConnected(ErrorCode err)
771  {
772  if (!err)
773  {
774  Base::listenForActions(
775  boost::bind(&This::onActionReceived,
776  shared_from_this(), _1, _2));
777  }
778 
779  if (boost::shared_ptr<IDeviceServiceEvents> cb = m_callbacks.lock())
780  {
781  cb->onConnected(err);
782  }
783  else
784  assert(!"callback is dead or not initialized");
785  }

788  /// @brief The "action sent" handler.
789  /**
790  @param[in] err The error code.
791  @param[in] jaction The sent action.
792  */
793  void onActionSent(ErrorCode err, json::Value const& jaction)
794  {
795  if (err)
796  {
797  // notify the user if we cannot send action!
798  done(err, jaction);
799  }
800  }

802 
803  /// @brief The "action received" handler.
804  /**
805  @param[in] err The error code.
806  @param[in] jaction The received action.
807  */
808  void onActionReceived(ErrorCode err, json::Value const& jaction)
809  {
810  if (!err)
811  {
812  const String action = jaction["action"].asString();
813 
814  if (boost::iequals(action, "server/info"))
815  {
816  if (!boost::iequals(jaction["status"].asString(), "success"))
817  err = boost::asio::error::fault; // TODO: useful error code
818  else if (boost::shared_ptr<IDeviceServiceEvents> cb = m_callbacks.lock())
819  {
820  const json::Value jinfo = jaction["info"];
821 
822  ServerInfo info;
823  info.api_version = jinfo["apiVersion"].asString();
824  info.timestamp = jinfo["serverTimestamp"].asString();
825  info.alternativeUrl = jinfo["restServerUrl"].asString();
826 
827  cb->onServerInfo(err, info);
828  }
829  }
830 
831  else if (boost::iequals(action, "device/save")) // got "register device" or "update device data" response
832  {
833  if (!boost::iequals(jaction["status"].asString(), "success"))
834  err = boost::asio::error::fault; // TODO: useful error code
835  }
836 
837  else if (boost::iequals(action, "device/get")) // got "get device data" response
838  {
839  if (!boost::iequals(jaction["status"].asString(), "success"))
840  err = boost::asio::error::fault; // TODO: useful error code
841  else
842  {
843  // update device's data
844  json::Value const& jdev = jaction["device"];
845  if (DevicePtr device = findDeviceById(jdev["id"].asString()))
846  Serializer::fromJson(jdev, device);
847  }
848  }
849 
850  else if (boost::iequals(action, "command/update")
851  || boost::iequals(action, "command/subscribe")
852  || boost::iequals(action, "command/unsubscribe"))
853  {
854  if (!boost::iequals(jaction["status"].asString(), "success"))
855  err = boost::asio::error::fault; // TODO: useful error code
856  }
857 
858  else if (boost::iequals(action, "command/insert"))
859  {
860  if (DevicePtr device = findDeviceById(jaction["deviceGuid"].asString()))
861  if (boost::shared_ptr<IDeviceServiceEvents> cb = m_callbacks.lock())
862  {
863  CommandPtr command = Command::create();
864  Serializer::fromJson(jaction["command"], command);
865  cb->onInsertCommand(err, device, command);
866  }
867  }
868  }
869  else
870  {
871  // report errors via 'onConnected' event!?
872  if (boost::shared_ptr<IDeviceServiceEvents> cb = m_callbacks.lock())
873  {
874  cb->onConnected(err);
875  }
876  }
877 
878  done(err, jaction);
879  }
880 
881 
882  /// @brief Finish the action.
883  /**
884  This method calls the corresponding callback functor if any.
885 
886  @param[in] err The error code.
887  @param[in] jaction The action to finish.
888  */
889  void done(ErrorCode err, json::Value const& jaction)
890  {
891  const UInt64 reqId = jaction["requestId"].asUInt();
892  std::map<UInt64, ActionCallback>::iterator it = m_actions.find(reqId);
893  if (it != m_actions.end())
894  {
895  it->second(err); // callback
896  m_actions.erase(it);
897  }
898  }
899 
900 
901  /// @brief Find device by UUID.
902  /**
903  @param[in] id The device identifer.
904  @return The device or NULL.
905  */
906  DevicePtr findDeviceById(String const& id) const
907  {
908  std::set<DevicePtr>::const_iterator i = m_devices.begin();
909  std::set<DevicePtr>::const_iterator e = m_devices.end();
910  for (; i != e; ++i)
911  {
912  DevicePtr device = *i;
913  if (device->id == id)
914  return device;
915  }
916 
917  return DevicePtr(); // not found
918  }
919 
920 private:
921  boost::weak_ptr<IDeviceServiceEvents> m_callbacks;
922  std::set<DevicePtr> m_devices;
923 
924 private:
925  typedef boost::function1<void, ErrorCode> ActionCallback;
926  std::map<UInt64, ActionCallback> m_actions;
927  UInt64 m_requestId;
928 };
