686  ErrorCode verifyTaskResponse(http::Client::TaskPtr task, const char *hint)

689

693  {

695  }
696  else

700  }
701  }
702  else if (!task->response)
703  {
704  HIVELOG_WARN(m_log, "failed to get \"" << hint << "\": no response");
705  err = boost::asio::error::fault; // TODO: useful error code
706  }
707  else if (!task->response->isStatusSuccessful())
708  {
709  HIVELOG_WARN(m_log, "failed to get \"" << hint
710  << "\": HTTP status: " << task->response->getStatusCode()
711  << " " << task->response->getStatusPhrase());
712  err = boost::asio::error::fault; // TODO: useful error code
713  }
714
