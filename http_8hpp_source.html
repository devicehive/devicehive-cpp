297  /// @brief Parse a string representation of an URL.
298  /**
299  @param url The URL string to parse.
300  @throws std::invalid_argument If URL string is invalid.
301  */
302  void parse(const char* url)
303  {
304  // protocol
305  if (const char* p = strstr(url, "://"))
306  {
307  const size_t len = (p - url);
308 
309  m_proto.assign(url, p);
310  //m_proto = misc::url_decode(m_proto);
311  boost::algorithm::to_lower(m_proto);
312 
313  url += len + 3;
314  }
315  else
316  throw std::invalid_argument("no protocol scheme");
317 
318  { // user information
319  const size_t len = strcspn(url, "@:[/?#");
320  if (url[len] == '@')
321  {
322  m_user.assign(url, url + len);
323  m_user = misc::url_decode(m_user);
324  url += len + 1;
325  }
326  else if (url[len] == ':')
327  {
328  const size_t len2 = strcspn(url+len, "@/?#");
329  if (url[len+len2] == '@')
330  {
331  m_user.assign(url, url + len+len2);
332  m_user = misc::url_decode(m_user);
333  url += len+len2 + 1;
334  }
335  }
336  }
337 
338  // host
339  if (url[0] == '[')
340  {
341  throw std::invalid_argument("IPv6 address not supported");
342  }
343  else
344  {
345  const size_t len = std::strcspn(url, ":/?#");
346  m_host.assign(url, url + len);
347  //m_host = misc::url_decode(m_host);
348  url += len;
349  }
350 
351  // port
352  if (url[0] == ':')
353  {
354  const size_t len = std::strcspn(++url, "/?#");
355  if (!len)
356  throw std::invalid_argument("no port number provided");
357 
358  m_port.assign(url, url + len);
359  for (size_t i = 0; i < len; ++i)
360  {
361  if (!misc::is_digit(m_port[i]))
362  throw std::invalid_argument("invalid port number");
363  }
364 
365  url += len;
366  }
367 
368  // path
369  if (url[0] == '/')
370  {
371  const size_t len = std::strcspn(url, "?#");
372  m_path.assign(url, url + len);
373  m_path = misc::url_decode(m_path);
374  url += len;
375  }
376  else
377  m_path = "/";
378 
379  // query
380  if (url[0] == '?')
381  {
382  const size_t len = std::strcspn(++url, "#");
383  m_query.assign(url, url + len);
384  m_query = misc::url_decode(m_query);
385  url += len;
386  }
387 
388  // fragment
389  if (url[0] == '#')
390  {
391  m_fragment.assign(++url);
392  m_fragment = misc::url_decode(m_fragment);
393  }
394  }

3055  int &vmajor, int &vminor, int &status, String &reason)
3056  {

3058  if (first == last || *first++ != 'H')

3060  if (first == last || *first++ != 'T')
3061  return false;
3062  if (first == last || *first++ != 'T')
3063  return false;
3064  if (first == last || *first++ != 'P')
3065  return false;
3066  if (first == last || *first++ != '/')
3067  return false;
3068 
3069  // parse major version "XXX."
3070  if (first == last || !misc::is_digit(*first))
3071  return false;

3073  while (first != last && misc::is_digit(*first))
3074  vmajor = 10*vmajor + misc::dec2int(*first++);
3075  if (first == last || *first++ != '.')
3076  return false;
3077 
3078  // parse minor version "XXX "
3079  if (first == last || !misc::is_digit(*first))
3080  return false;
3081  vminor = misc::dec2int(*first++);
3082  while (first != last && misc::is_digit(*first))
3083  vminor = 10*vminor + misc::dec2int(*first++);
3084  if (first == last || *first++ != ' ')
3085  return false;

3088  if (first == last || !misc::is_digit(*first))
