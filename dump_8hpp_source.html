127 OStream& int2hex(OStream &os, IntX x)
128 {
129  const size_t N = 2*sizeof(IntX);
130  for (size_t i = 0; i < N; ++i)

246 /// @brief Dump a binary data to an output stream in *ASCII* format.
247 /**
248 @param[in,out] os The output stream.
249 @param[in] first The begin of the binary data.
250 @param[in] last The end of the binary data.
251 @param[in] bad The replacement for non-ASCII characters.
252 @return The output stream.
253 */
254 template<typename In>
255 OStream& ascii(OStream &os, In first, In last, int bad = '.')
256 {
257  for (; first != last; ++first)
258  {
259  const int b = int(*first); // one byte
260  if (misc::is_char(b) && !misc::is_ctl(b))
261  os.put(b);
262  else
263  os.put(bad);
264  }
265 
266  return os;
267 }
