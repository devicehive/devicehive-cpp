157  /// @brief Assign the frame content.
158  /**
159  This method searches frame signature, veryfies the checksum
160  and assigns the frame content.
161 
162  @param[in] first The begin of input data.
163  @param[in] last The end of input data.
164  @param[out] n_skip The number of bytes processed.
165  @param[out] result The parse result. May be NULL.
166  @return Parsed frame or NULL.
167  */
168  template<typename In>
169  static SharedPtr parseFrame(In first, In last, size_t &n_skip, ParseResult *result)
170  {
171  ParseResult res = RESULT_INCOMPLETE;
172  SharedPtr frame; // empty frame
173 
174  // search for signature
175  for (; first != last; ++first, ++n_skip)
176  {
177  if (UInt8(*first) == SIGNATURE)
178  break;
179  }
180 
181  const size_t buf_len = std::distance(first, last);
182  if (HEADER_LEN+0+FOOTER_LEN <= buf_len) // can parse header
183  {
184  const In s_first = first;
185 
186  first += SIGNATURE_LEN; // skip signature
187  const int len_msb = UInt8(*first++);
188  const int len_lsb = UInt8(*first++);
189  const size_t len = (len_msb<<8) | len_lsb;
190 
191  if (HEADER_LEN+len+FOOTER_LEN <= buf_len) // can parse whole frame
192  {
193  if (checksum(first, first+(len+CHECKSUM_LEN)) != 0x00)
194  {
195  // bad checksum
196  res = RESULT_BADCHECKSUM;
197  n_skip += SIGNATURE_LEN; // skip signature
198  }
199  else
200  {
201  // assign the frame content
202  frame.reset(new Frame());
203  frame->m_content.assign(s_first,
204  s_first+(HEADER_LEN+len+FOOTER_LEN));
205  n_skip += HEADER_LEN+len+FOOTER_LEN;
206  res = RESULT_SUCCESS;
207  }
208  }
209  //else incomplete
210  }
211  // else incomplete
212 
213  if (result)
214  *result = res;
215  return frame;
216  }
217 
218 
219  /// @brief Assign the frame content from stream buffer.
220  /**
221  This method searches frame signature, veryfies the checksum
222  and assigns the frame content.
223 
224  @param[in,out] sb The input data buffer.
225  @param[out] result The parse result. May be NULL.
226  @return Parsed frame or NULL.
227  */
228  static SharedPtr parseFrame(boost::asio::streambuf & sb, ParseResult *result)
229  {
230  size_t n_skip = 0; // number of bytes to skip
231  const boost::asio::streambuf::const_buffers_type bufs = sb.data();
232  SharedPtr frame = parseFrame(boost::asio::buffers_begin(bufs),
233  boost::asio::buffers_end(bufs), n_skip, result);
234  sb.consume(n_skip);
235  return frame;
236  }

254  /// @brief Calculate the checksum byte.
255  /**
256  @param[in] first The begin of frame data payload.
257  @param[in] last The end of frame data payload.
258  @return The checksum byte.
259  */
260  template<typename In>
261  static UInt8 checksum(In first, In last)
262  {
263  int cs = 0;
264  for (; first != last; ++first)
265  cs += UInt8(*first);
266  return UInt8(0xFF - (cs&0xFF));
267  }

270  /// @brief Initialize frame content from payload data.
271  /**
272  @param[in] payload The frame payload data.
273  */
274  void init(String const& payload)
275  {
276  const size_t len = payload.size();
277  assert(len <= 64*1024 && "frame data payload too big");
278 
279  m_content.reserve(HEADER_LEN+len+FOOTER_LEN);
280 
281  // header (signature+length)
282  m_content.push_back(SIGNATURE);
283  m_content.push_back((len>>8)&0xFF); // length, MSB
284  m_content.push_back((len)&0xFF); // length, LSB
285 
286  // payload
287  m_content.insert(m_content.end(),
288  payload.begin(), payload.end());
289 
290  // checksum
291  m_content.push_back(
292  checksum(payload.begin(),
293  payload.end()));
294  }
