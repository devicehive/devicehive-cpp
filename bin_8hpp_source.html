806 /// @brief The simple frame example.
807 /**
808 The main aim of this class is to demonstrate frame prototype for all other formats.
809 
810 The frame format is quite simple:
811 
812 | field | size in bytes
813 |-----------|--------------
814 | signature | 1
815 | length | 1
816 | intent | 2 (little-endian)
817 | payload | N
818 | checksum | 1
819 */
820 class SimpleFrame:
821  public FrameContent
822 {
823 public:
824 
825  /// @brief Constants.
826  enum Const
827  {
828  SIGNATURE = 0xFC, ///< @brief The signature.
829  };
830 
831 private:
832 
833  /// @brief Various lengths.
834  enum Length
835  {
836  SIGNATURE_LEN = 1, ///< @brief The "signature" field length in bytes.
837  LENGTH_LEN = 1, ///< @brief The "length" field length in bytes.
838  INTENT_LEN = 2, ///< @brief The "intent" field length in bytes.
839  CHECKSUM_LEN = 1, ///< @brief The "checksum" field length in bytes.
840 
841  /// @brief The total header length in bytes.
842  HEADER_LEN = SIGNATURE_LEN + LENGTH_LEN + INTENT_LEN,
843 
844  /// @brief The total footer length in bytes.
845  FOOTER_LEN = CHECKSUM_LEN
846  };
847 
848 protected:
849 
850  /// @brief The default constructor.
851  /**
852  Constructs the empty frame.
853  */
854  SimpleFrame()
855  {}
856 
857 public:
858 
859  /// @brief The shared pointer type.
860  typedef boost::shared_ptr<SimpleFrame> SharedPtr;
861 
862 
863  /// @brief Construct the frame from the payload.
864  /**
865  @param[in] intent The message intent.
866  @param[in] payload The data payload.
867  Should be byte container with `begin()` and `end()` methods.
868  For example std::vector<char> or hive::String.
869  @return The new frame instance.
870  */
871  template<typename PayloadT>
872  static SharedPtr create(int intent, PayloadT const& payload)
873  {
874  SharedPtr pthis(new SimpleFrame());
875  pthis->init(intent, payload);
876  return pthis;
877  }
878 
879 public:
880 
881  /// @brief Parse the payload from the frame.
882  /**
883  @param[out] payload The data payload to store.
884  Should be byte container with `assign()` method.
885  For example std::vector<char> or hive::String.
886  @return `true` if data payload successfully parsed.
887  */
888  template<typename PayloadT>
889  bool getPayload(PayloadT &payload) const
890  {
891  if (HEADER_LEN+FOOTER_LEN <= m_content.size())
892  {
893  payload.assign(
894  m_content.begin()+HEADER_LEN, // skip signature+length+intent
895  m_content.end()-FOOTER_LEN); // skip checksum
896  return true;
897  }
898 
899  return false; // empty
900  }
901 
902 public:
903 
904  ///@brief The frame parse result.
905  enum ParseResult
906  {
907  RESULT_SUCCESS, ///< @brief Frame succcessfully parsed.
908  RESULT_INCOMPLETE, ///< @brief Need more data.
909  RESULT_BADCHECKSUM ///< @brief Bad checksum found.
910  };
911 
912 
913  /// @brief Assign the frame content.
914  /**
915  This method searches frame signature, veryfies the checksum
916  and assigns the frame content.
917 
918  @param[in] first The begin of input data.
919  @param[in] last The end of input data.
920  @param[out] n_skip The number of bytes processed.
921  @param[out] result The parse result. May be NULL.
922  @return Parsed frame or NULL.
923  */
924  template<typename In>
925  static SharedPtr parseFrame(In first, In last, size_t &n_skip, ParseResult *result)
926  {
927  ParseResult res = RESULT_INCOMPLETE;
928  SharedPtr frame; // empty frame
929 
930  // search for signature
931  for (; first != last; ++first, ++n_skip)
932  {
933  if (UInt8(*first) == SIGNATURE)
934  break;
935  }
936 
937  const size_t buf_len = std::distance(first, last);
938  if (HEADER_LEN+FOOTER_LEN <= buf_len) // can parse header
939  {
940  const In s_first = first;
941 
942  first += SIGNATURE_LEN; // skip signature
943  const size_t len = UInt8(*first++);
944  first += INTENT_LEN; // skip intent
945 
946  if (size_t(HEADER_LEN+len+FOOTER_LEN) <= buf_len) // can parse whole frame
947  {
948  if (checksum(s_first, s_first+(HEADER_LEN+len+FOOTER_LEN)) != 0x00)
949  {
950  // bad checksum
951  res = RESULT_BADCHECKSUM;
952  n_skip += SIGNATURE_LEN; // skip signature
953  }
954  else
955  {
956  // assign the frame content
957  frame.reset(new SimpleFrame());
958  frame->m_content.assign(s_first,
959  s_first+(HEADER_LEN+len+FOOTER_LEN));
960  n_skip += HEADER_LEN+len+FOOTER_LEN;
961  res = RESULT_SUCCESS;
962  }
963  }
964  //else incomplete
965  }
966  // else incomplete
967 
968  if (result)
969  *result = res;
970  return frame;
971  }

974  /// @brief Assign the frame content from stream buffer.
975  /**
976  This method searches frame signature, veryfies the checksum
977  and assigns the frame content.
978 
979  @param[in,out] sb The input data buffer.
980  @param[out] result The parse result. May be NULL.
981  @return Parsed frame or NULL.
982  */
983  static SharedPtr parseFrame(boost::asio::streambuf &sb, ParseResult *result)
984  {

986  const boost::asio::streambuf::const_buffers_type bufs = sb.data();
987  SharedPtr frame = parseFrame(boost::asio::buffers_begin(bufs),
988  boost::asio::buffers_end(bufs), n_skip, result);
989  sb.consume(n_skip);
990  return frame;
991  }

993 public:
994 
995  /// @brief Get the frame intent.
996  /**
997  @return The frame intent or `-1` if it's unknown.
998  */
999  int getIntent() const
1000  {
1001  if (HEADER_LEN+0+FOOTER_LEN <= m_content.size())
1002  {
1003  size_t offset = SIGNATURE_LEN+LENGTH_LEN;
1004 
1005  const int id_lsb = m_content[offset++];
1006  const int id_msb = m_content[offset++];
1007

1009  }
1010 
1011  return -1; // unknown
1012  }

1015 
1016  /// @brief Calculate the checksum byte.
1017  /**
1018  @param[in] first The begin of frame data payload.
1019  @param[in] last The end of frame data payload.
1020  @return The checksum byte.
1021  */
1022  template<typename In>
1023  static UInt8 checksum(In first, In last)
1024  {
1025  int cs = 0;
1026  for (; first != last; ++first)
1027  cs += UInt8(*first);
1028  return UInt8(0xFF - (cs&0xFF));
1029  }

1033  /**
1034  @param[in] intent The payload intent.
1035  @param[in] payload The frame payload data.
1036  */
1037  template<typename PayloadT>
1038  void init(int intent, PayloadT const& payload)
1039  {
1040  const size_t len = payload.size();
1041  assert(len < 255 && "frame data payload too big");
1042 
1043  m_content.reserve(HEADER_LEN+len+FOOTER_LEN);
1044 
1045  // header (signature+length+intent)
1046  m_content.push_back(SIGNATURE); // signature
1047  m_content.push_back(len&0xFF); // length
1048  m_content.push_back((intent)&0xFF); // intent, LSB
1049  m_content.push_back((intent>>8)&0xFF); // intent, MSB
1050 
1051  // insert payload
1052  m_content.insert(m_content.end(),
1053  payload.begin(), payload.end());
1054 
1055  // checksum
1056  m_content.push_back(checksum(
1057  m_content.begin(),
1058  m_content.end()));
1059  }
