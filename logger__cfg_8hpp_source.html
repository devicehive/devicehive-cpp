20 /// @brief The log configuration.
21 /**
22 The main configuration should be an JSON object with the following members:
23 
24 ~~~{.js}
25 configuration: {
26  targets: {
27  target1: { ... }
28  target2: { ... }
29  ...
30  }
31 
32  loggers: {
33  logger1: { ... }
34  logger2: { ... }
35  ...
36  }
37 }
38 ~~~
39 
40 Each *target* description shoud be a valid JSON object with the mandatory
41 `type` member which could be one of the following:
42  - `file` for file targets
43  - `stdout` or `console` for `std::cout` target
44  - `stderr` for `std::cerr` target
45  - `null` for **NULL** target
46 
47 Each target supports the following common parameters:
48  - `minimumLevel` is the minimum logging level for that target,
49  see below for possible values
50  - `format` is the custom format string,
51  see hive::log::Format for more details
52 
53 The `file` target also supports the following parameters:
54  - `fileName` is mandatory log file name
55  - `autoFlushLevel` is the *auto-flush* level, see below for possible values
56  - `maxFileSize` is the maximum log file size in bytes,
57  `K`, `M` and `G` suffixes supported
58  - `numOfBackups` is the maximum number of backup log files.
59 
60 The *logger* supports the following parameters:
61  - `level` is the logging level, see below for possible values
62  - `targets` is the array of targets for this logger
63 
64 The logging level can be one of:
65 
66 | string identifier | logging level |
67 |-----------------------|-------------------------------|
68 | `"ALL"` | hive::log::LEVEL_ALL |
69 | `"TRACE"` | hive::log::LEVEL_TRACE |
70 | `"DEBUG"` | hive::log::LEVEL_DEBUG |
71 | `"INFO"` | hive::log::LEVEL_INFO |
72 | `"WARN"` | hive::log::LEVEL_WARN |
73 | `"ERROR"` | hive::log::LEVEL_ERROR |
74 | `"FATAL"` | hive::log::LEVEL_FATAL |
75 | `"OFF"` \n `"NO"` | hive::log::LEVEL_OFF |
76 | `"ASPARENT"` \n `"-"` | hive::log::LEVEL_AS_PARENT |
77 
78 Example:
79 
80 ~~~{.js}
81 {
82  targets: {
83  console: {
84  type: "console"
85  minimumLevel: "INFO"
86  format: "%L %N %M\n"
87  }
88  myfile: {
89  type: "file"
90  fileName: "test.log"
91  maxFileSize: "1M"
92  numOfBackups: 1
93  }
94  }
95 
96  loggers: {
97  "/": {
98  level: "TRACE"
99  targets: ["myfile", "console"]
100  }
101  "/API": {
102  level: "DEBUG"
103  targets: ["myfile"]
104  }
105  }
106 }
107 ~~~
108 */

136  /// @brief Apply parsed configuration.
137  void apply() const
138  {
139  typedef std::map<hive::String, LoggerInfo>::const_iterator Iterator;
140  Iterator i = m_loggers.begin();
141  Iterator e = m_loggers.end();
142  for (; i != e; ++i)
143  {
144  hive::String const& name = i->first;
145  LoggerInfo const& info = i->second;
146  using namespace hive::log;
147 
148  Logger logger(name);
149  if (1 < info.targets.size())
150  {
151  Target::Tie::SharedPtr tie = Target::Tie::create();
152  for (size_t i = 0; i < info.targets.size(); ++i)
153  tie->add(info.targets[i]); // default level!
154  logger.setTarget(tie);
155  }
156  else if (!info.targets.empty())
157  logger.setTarget(info.targets.front());
158 
159  if (info.level != -1)
160  logger.setLevel(Level(info.level));
161  }
162  }

190  @return The new log target.
191  @throw std::runtime_error in case of error.
192  */
193  static TargetPtr makeTarget(JsonValue const& jtarget)
194  {
195  using namespace hive::log;
196 
197  try
198  {
199  hive::String ttype;
200  TargetPtr target;
201 
202  // get target type
203  if (jtarget.isNull())
204  ttype = "null";
205  else if (jtarget.isString())
206  ttype = jtarget.asString();
207  else if (jtarget.isObject())
208  ttype = jtarget["type"].asString();
209  else
210  throw std::runtime_error("invalid target configuration");
211 
212  if (boost::iequals(ttype, "file"))
213  {
214  if (!jtarget.isObject())
215  throw std::runtime_error("invalid file target configuration");
216 
217  const hive::String fileName = jtarget["fileName"].asString();
218  if (fileName.empty())
219  throw std::runtime_error("not file name provided");
220  Target::File::SharedPtr file = Target::File::create(fileName);
221 
222  // auto-flush level
223  const hive::String autoFlushLevel = jtarget["autoFlushLevel"].asString();
224  if (!autoFlushLevel.empty())
225  file->setAutoFlushLevel(parseLevel(autoFlushLevel));
226 
227  // maximum file size
228  const hive::String maxFileSize = jtarget["maxFileSize"].asString();
229  if (!maxFileSize.empty())
230  file->setMaxFileSize(parseFileSize(maxFileSize));
231 
232  // number of backups
233  file->setNumberOfBackups(jtarget["numOfBackups"].asUInt16());
234 
235  target = file;
236  }
237  else if (boost::iequals(ttype, "stdout")
238  || boost::iequals(ttype, "console"))
239  {
240  target = Target::Stream::createStdout();
241  }
242  else if (boost::iequals(ttype, "stderr"))
243  {
244  target = Target::Stream::createStderr();
245  }
246  else if (boost::iequals(ttype, "null"))
247  {
248  target = Target::create(); // NULL
249  }
250 
251  // common properties
252  if (target && jtarget.isObject())
253  {
254  // minimum logging level
255  const hive::String minimumLevel = jtarget["minimumLevel"].asString();
256  if (!minimumLevel.empty())
257  target->setMinimumLevel(parseLevel(minimumLevel));
258 
259  // custom format
260  const hive::String format = jtarget["format"].asString();
261  if (!format.empty())
262  target->setFormat(Format::create(format));
263  }
264 
265  // unknown target
266  if (!target)
267  {
268  hive::OStringStream ess;
269  ess << "\"" << ttype << "\" is unknown log target type";
270  throw std::runtime_error(ess.str().c_str());
271  }
272 
273  return target;
274  }
275  catch (std::exception const& ex)
276  {
277  hive::OStringStream ess;
278  ess << "cannot create target: "
279  << ex.what();
280  throw std::runtime_error(ess.str().c_str());
281  }
282  }

351  /// @brief Parse the log level.
352  /**
353  @param[in] level The logging level name.
354  @return The logging level.
355  @throw std::runtime_error in case of error.
356  */
357  static hive::log::Level parseLevel(hive::String const& level)
358  {
359  using namespace hive;
360 
361  if (boost::iequals(level, "ALL"))
362  return log::LEVEL_ALL;
363  else if (boost::iequals(level, "TRACE"))
364  return log::LEVEL_TRACE;
365  else if (boost::iequals(level, "DEBUG"))
366  return log::LEVEL_DEBUG;
367  else if (boost::iequals(level, "INFO"))
368  return log::LEVEL_INFO;
369  else if (boost::iequals(level, "WARN"))
370  return log::LEVEL_WARN;
371  else if (boost::iequals(level, "ERROR"))
372  return log::LEVEL_ERROR;
373  else if (boost::iequals(level, "FATAL"))
374  return log::LEVEL_FATAL;
375  else if (boost::iequals(level, "OFF") || boost::iequals(level, "NO"))
376  return log::LEVEL_OFF;
377  else if (boost::iequals(level, "ASPARENT") || boost::iequals(level, "-"))
378  return log::LEVEL_AS_PARENT;
379  else // unknown level
380  {
381  hive::OStringStream ess;
382  ess << "\"" << level << "\" is unknown logging level";
383  throw std::runtime_error(ess.str().c_str());
384  }
385  }

388  /// @brief Parse the file size.
389  /**
390  @param[in] fileSize The file size string.
391  May be siffuxed with `K`, `M` or `G`.
392  @return The file size in bytes.
393  @throw std::runtime_error in case of error.
394  */
395  static size_t parseFileSize(hive::String const& fileSize)
396  {
397  hive::IStringStream iss(fileSize);
398 
399  double size = 0;
400  if (!(iss >> size))
401  throw std::runtime_error("cannot parse size");
402 
403  hive::String suffix; // optional
404  if ((iss >> suffix) && !suffix.empty())
405  {
406  if (boost::iequals(suffix, "K"))
407  size *= 1024;
408  else if (boost::iequals(suffix, "M"))
409  size *= 1024*1024;
410  else if (boost::iequals(suffix, "G"))
411  size *= 1024*1024*1024;
412  else // unknown suffix
413  {
414  hive::OStringStream ess;
415  ess << "\"" << suffix << "\" is unknown file size suffix";
416  throw std::runtime_error(ess.str().c_str());
417  }
418  }
419 
420  if (!(iss >> std::ws).eof())
421  throw std::runtime_error("invalid file size");
422 
423  if (size < std::numeric_limits<size_t>::min()
424  || std::numeric_limits<size_t>::max() < size)
425  {
426  throw std::runtime_error("invalid file size - out of range");
427  }
428 
429  return size_t(size);
430  }
