589  /// @brief Handle the incomming message.
590  /**
591  @param[in] intent The message intent.
592  @param[in] data The custom message data.
593  */
594  void handleGatewayMessage(int intent, json::Value const& data)
595  {
596  HIVELOG_INFO(m_log, "process received frame #" << intent << " data: " << data);
597 
598  if (intent == gateway::INTENT_REGISTRATION_RESPONSE)
599  {
600  if (m_device)
601  m_service->asyncUnsubscribeFromCommands(m_device);
602  m_gw.handleRegisterResponse(data);
603  createDevice(data);
604  m_service->asyncRegisterDevice(m_device);
605  }
606 
607  else if (intent == gateway::INTENT_REGISTRATION2_RESPONSE)
608  {
609  if (m_device)
610  m_service->asyncUnsubscribeFromCommands(m_device);
611  json::Value jdev = json::fromStr(data["json"].asString());
612  m_gw.handleRegister2Response(jdev);
613  createDevice(jdev);
614  m_service->asyncRegisterDevice(m_device);
615  }
616 
617  else if (intent == gateway::INTENT_COMMAND_RESULT_RESPONSE)
618  {
619  if (m_device)
620  {
621  HIVELOG_DEBUG(m_log, "got command result");
622 
623  devicehive::CommandPtr command = devicehive::Command::create();
624  command->id = data["id"].asUInt();
625  command->status = data["status"].asString();
626  command->result = data["result"];
627 
628  m_service->asyncUpdateCommand(m_device, command);
629  }
630  else
631  HIVELOG_WARN_STR(m_log, "got command result before registration, ignored");
632  }
633 
634  else if (intent >= gateway::INTENT_USER)
635  {
636  if (m_device)
637  {
638  String name = m_gw.findNotificationNameByIntent(intent);
639  if (!name.empty())
640  {
641  HIVELOG_DEBUG(m_log, "got notification");
642 
643  devicehive::NotificationPtr notification = devicehive::Notification::create(name, data["parameters"]);
644 
645  if (m_deviceRegistered)
646  {
647  m_service->asyncInsertNotification(m_device, notification);
648  }
649  else
650  {
651  HIVELOG_DEBUG_STR(m_log, "device is not registered, notification delayed");
652  m_pendingNotifications.push_back(notification);
653  }
654  }
655  else
656  HIVELOG_WARN(m_log, "unknown notification: " << intent << ", ignored");
657  }
658  else
659  HIVELOG_WARN_STR(m_log, "got notification before registration, ignored");
660  }
661 
662  else
663  HIVELOG_WARN(m_log, "invalid intent: " << intent << ", ignored");
664  }

731 This simple example aims to demonstrate usage of binary DeviceHive protocol
732 from C++ application.
733 
734 In terms of DeviceHive framework this example is **Gateway** implementation:
735 
736 ![DeviceHive framework](@ref preview2.png)
737 
738 Our target device is one of small microcontrollers (like [MSP430] development
739 board or [PIC16F88] based board) which runs microcode and manages the following
740 peripheral:
741  - simple LED, allowing user to switch it *ON* or *OFF*
742  - simple push button, providing user with custom asynchronous events
743 
744 Because of limited resources it's impossible to use RESTful protocol on
745 such small devices. So device uses binary protocol to communicate with an
746 intermediate node - gateway. Gateway is used to convert binary messages
747 to the JSON messages of RESTful protocol and visa versa.
748 
749 The full source code available in examples/simple_gw.hpp file.
750 
751 It's highly recommended to study @ref page_simple_dev first.
752 
753 Please see @ref page_start to get information about how to prepare toolchain
754 and build the *boost* library.

758 ===========
759 
760 Application is a console program and it's reperented by an instance of
761 simple_gw::Application. This instance can be created using
762 simple_gw::Application::create() factory method. Application does all
763 its useful work in simple_gw::Application::run() method.
764 
765 ~~~{.cpp}
766 using namespace simple_gw;

768 {

771 }

775 Binary message layout
776 =====================
777 
778 Each binary message has it's own layout - the binary data format - according
779 to binary protocol specification. Simple, layout is the list of data fields
780 of various types (integers, string, etc.) and it's names.
781 Layout also may be treated as a binary<->JSON conversion rule.
782 
783 Message layout is represented by gateway::Layout class. Besides system layouts
784 wich have known format, each endpoint device should provide layout for
785 device-specific messages during registration procedure.
786 
787 Each message is identified by unique intent number. The gateway::LayoutManager
788 class manages such intent numbers and corresponding layouts.
789 
790 
791 Control flow
792 ------------
793 
794 At the start we should send registration request to the target device connected
795 via serial cable. Once we got registration response we register our device on
796 the server using simple_gw::Application::asyncRegisterDevice() method.
797 
798 If registration is successful we start listening for commands from the
799 DeviceHive server using simple_gw::Application::asyncPollCommands() method.
800 Once we received command we convert it from JSON to binary format and send
801 converted command to the device via serial port.
802 
803 At the same time we listen for notifications from serial port. Once we got
804 notification from device we convert it from binary to JSON format and send
805 converted notification to the DeviceHive server.
806
